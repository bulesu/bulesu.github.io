    # 对象属性的操作

    # 1.1属性的读取

    读取对象的属性，有两种方法，一种是使用点运算符，还有一种是使用方括号运算符。

    var obj = {
  p: 'Hello World'
};

obj.p // "Hello World"
obj['p'] // "Hello World"


var foo = 'bar';
var obj = {
  foo: 1,
  bar: 3
};

obj.foo  // 1
obj[foo]  // 3

// 引用对象obj的foo属性时，如果使用点运算符，foo就是字符串；
// 如果使用方括号运算符，但是不使用引号，那么foo就是一个变量，指向字符串bar。

    # 2.2对象的方法

      var Obj = { p : 1 }
  Obj.delete(p)   //delete()方法 console.log(Obj.p)  // undefined

  Obj.delete(a) //true
  //注意，删除一个不存在的属性，delete不报错，而且返回true。

  length:用于返回字符串的长度，用法与数组一样；

    # 2.4属性方法 Object.defineProperty()

    # 直接在一个对象上定义一个新属性，或者修改一个对象的现有属性，并返回此对象

    Object.defineProperty(obj, prop, descriptor) // 方法

参数：obj===>要定义属性的对象
prop===>要定义或修改的属性的名称或Symbol
descriptor===>要定义或修改的属性描述符

描述符：分为数据描述符和存取描述符，一个描述符只能是两者的其中之一，不能同时是两者



let obj = {}
Object.defineProperty(obj, 'name', {
  get: function() {
    console.log('调用了get方法')
  },
  set: function(newVal) { 
    console.log('调用了set方法，方法的值是：' + newVal)
  }
})
obj.name // 打印：调用了get方法
obj.name = 'Sulr' // 打印：调用了set方法，方法的值是：Sulr

    # 简单实现一个vue里的双向绑定原理

    <body>
  <div id="app">
    <input id="a" type="text">
    <h1 id="b"></h1>
  </div>
  <script>
    const domA = document.getElementById('a')
    const domB = document.getElementById('b')
    let obj = {}
    let val = 'Sulr'
    Object.defineProperty(obj, 'name', {
      get: function() {
        console.log('get val:' + val)
        return val
      },
      set: function(newVal) {
        val = newVal
        console.log('set val:' + val)
        domA.value = val
        domB.innerHTML = val
      }
    })
    // 监听键盘按下
    domA.addEventListener('keyup', function(e) {
      obj.name = e.target.value
    })
  </script>
</body>
